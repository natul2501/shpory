Оператор switch

var day;
switch (new Date().getDay()) {
	case 6:
		day = "Сегодня суббота";
		break; 
	case 0:
		day = "Сегодня воскресенье";
		break; 
	default: 
		day = "Будем ждать выходных";
}
document.write(day);
